setupCorrelations() — STREAM-BASED WIRING (RXJS)

// Every behavior is a stream. Operators transform and combine streams.
// This function is the only place where correlations are defined.

function setupCorrelations(switchEl) {
  const { fromEvent, timer, of } = rxjs
  const { scan, switchMap, map, take, share } = rxjs

  // 1. Click stream → boolean machine-on stream (toggles each click)
  const machineOn$ = fromEvent(switchEl, 'click').pipe(
    scan(on => !on, false),
    share()
  )

  // 2. Machine-on stream → temperature stream
  //    When on: emit low/medium/high every 10 s; when off: reset to low
  const temperature$ = machineOn$.pipe(
    switchMap(on => on
      ? timer(0, 10000).pipe(take(3), map(t => LEVELS[t]))
      : of('low')
    ),
    share()
  )

  // 3. Derived stream: cooler on when temperature is high
  const coolerOn$ = temperature$.pipe(map(t => t === 'high'))

  // 4. Subscribe: connect each stream to its renderer
  machineOn$   .subscribe(on => { renderMachine(on); renderSwitch(on) })
  temperature$ .subscribe(t  =>   renderTemperature(t))
  coolerOn$    .subscribe(on =>   renderCooler(on))
  temperature$ .subscribe(t  =>   renderAlert(t))
}
