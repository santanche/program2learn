setupCorrelations() — STATE-BASED WIRING (VUE)

// Reactive state is the single source of truth.
// Watchers observe refs and react by updating other refs or triggering side-effects.
// This function is the only place where correlations are defined.

function setupCorrelations(switchOn, machineOn, tempLevel, coolerOn) {

  // 1. Switch position → machine on/off
  watch(switchOn, on => {
    machineOn.value = on
  })

  // 2. Machine on/off → start or stop temperature rise
  watch(machineOn, on => {
    on ? startTemperatureRise() : resetTemperature()
  })

  // 3. Temperature level → cooler on/off
  watch(tempLevel, level => {
    coolerOn.value = level === 'high'
  })
}
