setupCorrelations() — EVENT-BASED WIRING (HTMX)

// Each component communicates through named custom events.
// This function is the only place where correlations are defined.

function setupCorrelations() {

  // 1. Switch click → toggle machine state
  on('switch:clicked', () => {
    machineOn = !machineOn
    emit('machine:changed', { on: machineOn })
  })

  // 2. Machine change → start/stop temperature rise
  on('machine:changed', ({ detail }) => {
    detail.on ? startTemperatureRise() : resetTemperature()
  })

  // 3. Temperature change → cooler and alert light react
  on('temperature:changed', ({ detail }) => {
    emit('cooler:changed',  { on: detail.level === 'high' })
    emit('alert:changed',   { level: detail.level })
  })
}
