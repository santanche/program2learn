setupCorrelations() — STATE-BASED WIRING (VUE)

// Reactive state is the single source of truth.
// Computed properties derive new state; watchers trigger side-effects.
// This function is the only place where correlations are defined.

function setupCorrelations(machineOn, tempLevel) {

  // 1. Derived state: always computed from tempLevel (no explicit event needed)
  const coolerOn  = computed(() => machineOn.value && tempLevel.value === 'high')
  const thermoImg = computed(() => THERMOS[tempLevel.value])
  const alertImg  = computed(() => ALERTS[tempLevel.value])

  // 2. Side-effect correlation: machine on/off → start or stop temperature rise
  watch(machineOn, on => {
    on ? startTemperatureRise() : resetTemperature()
  })

  return { coolerOn, thermoImg, alertImg }
}
